interface FilterState {
  queues: string[];
  reviewers: string[];
  errorTypes: string[];
}

const [filters, setFilters] = useState<FilterState>({
  queues: [],
  reviewers: [],
  errorTypes: []
});

const applyFilters = () => {
  // 构建查询参数
  const params = new URLSearchParams();
  if (filters.queues.length) params.set('queues', filters.queues.join(','));
  if (filters.reviewers.length) params.set('reviewers', filters.reviewers.join(','));
  if (filters.errorTypes.length) params.set('error_types', filters.errorTypes.join(','));
  
  // 重新加载数据
  loadData(params.toString());
};
